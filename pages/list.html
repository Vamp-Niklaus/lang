List (std::list)

Doubly-linked list. Fast insertions and deletions anywhere.

Category: Sequence Container

Core Operations

#include <list>
std::list<int> l;

l.push_back(10);
l.push_front(5);
l.pop_back();
l.pop_front();

l.insert(l.begin(), 20);
l.erase(l.begin());

List-Specific Operations

l.splice(pos, other_list);    // Move elements from another list
l.remove(val);                // Remove all elements matching val
l.remove_if(pred);            // Remove elements matching predicate
l.unique();                   // Remove consecutive duplicates
l.merge(other_list);          // Merge two sorted lists
l.sort();                     // O(N log N) sort
l.reverse();                  // Reverse the list
