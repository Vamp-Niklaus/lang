Arrays & Sequential Storage

Fixed contiguous arrays vs dynamic structural arrays.

C++
#include <vector>

// Raw Fixed Array
int rawArr[5] = {1, 2, 3};

// Dynamic Array (Vector)
std::vector<int> vec = {1, 2, 3};
vec.push_back(4); // Append

// Sizing
int size = vec.size();

// Access
int first = vec[0];
JAVA
import java.util.ArrayList;

// Fixed Array
int[] rawArr = {1, 2, 3, 4, 5};

// Resizable List (Objects only)
ArrayList<Integer> list = new ArrayList<>();
list.add(1); 
list.add(2);

// Sizing
int size = list.size();

// Access
int first = list.get(0);
JAVASCRIPT
// Always dynamic and resizable
let arr = [1, 2, 3];

// Mutation operators
arr.push(4);    // Append item
arr.unshift(0); // Prepend item

// Sizing property
let size = arr.length;

// Element retrieval
let first = arr[0];
PYTHON
# Built-in Dynamic List type
arr = [1, 2, 3]

# Mutations
arr.append(4)   # Add to end
arr.insert(0, 0)# Positional insert

# Total Elements function
size = len(arr)

# Bracket access
first = arr[0]
last = arr[-1]  # Negative indexing
