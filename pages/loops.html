Loops & Control Flow

Iteration patterns over ranges and explicit counters.

C++
// Classic For Loop
for (int i = 0; i < 5; i++) {
    std::cout << i << "\n";
}

// While Loop
int w = 0;
while (w < 3) {
    w++;
}

// Range-based / For-Each
int arr[] = {10, 20, 30};
for (int val : arr) {
    std::cout << val;
}
JAVA
// Classic For Loop
for (int i = 0; i < 5; i++) {
    System.out.println(i);
}

// While Loop
int w = 0;
while (w < 3) {
    w++;
}

// Enhanced For loop
int[] arr = {10, 20, 30};
for (int val : arr) {
    System.out.println(val);
}
JAVASCRIPT
// Standard Counter Loop
for (let i = 0; i < 5; i++) {
    console.log(i);
}

// For...of (Iterates values)
const items = [10, 20, 30];
for (const val of items) {
    console.log(val);
}

// For...in (Iterates keys/index)
for (const index in items) {
    console.log(index);
}
PYTHON
# Range tracking (0 to 4)
for i in range(5):
    print(i)

# Custom range step (1 to 9 step 2)
for i in range(1, 10, 2):
    print(i)

# Element iterator
items = [10, 20, 30]
for val in items:
    print(val)

# Index paired iteration
for idx, val in enumerate(items):
    print(idx, val)
