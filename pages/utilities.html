Utilities

Useful helpers and string functions.

Pair & Tuple

#include <utility>
std::pair<int, int> p = {1, 2};

#include <tuple>
std::tuple<int, char, double> t = {1, 'a', 2.0};

String Functions

std::stoi("10"); // String to Int
std::to_string(10); // Number to String
std::getline(std::cin, str); // Read line

Swap & Reverse

std::swap(a, b); // O(1)
std::reverse(vec.begin(), vec.end()); // O(n)

Permutations

std::next_permutation(vec.begin(), vec.end());
std::prev_permutation(vec.begin(), vec.end());

Memory Set & Fill

std::fill(vec.begin(), vec.end(), val);
#include <cstring>
std::memset(ptr, -1, sizeof(ptr));

Math & Count

std::min(a, b);
std::max(a, b);
std::abs(val);
std::count(vec.begin(), vec.end(), val);
