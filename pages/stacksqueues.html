Stacks & Queues

LIFO and FIFO linear architectural access pipes.

C++
#include <stack>
#include <queue>

// LIFO Stack Container adapter
std::stack<int> st;
st.push(10);
int topVal = st.top();
st.pop();

// FIFO Queue Container adapter
std::queue<int> q;
q.push(1);
int frontVal = q.front();
q.pop();
JAVA
import java.util.Stack;
import java.util.LinkedList;
import java.util.Queue;

// Stack utility class
Stack<Integer> st = new Stack<>();
st.push(10);
int top = st.peek();
st.pop();

// Queue Interface implementation
Queue<Integer> q = new LinkedList<>();
q.add(1);     // Enqueue
int front = q.peek();
q.remove();   // Dequeue
