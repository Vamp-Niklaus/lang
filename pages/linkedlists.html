Linked Lists

Pointer-linked sequential memory structure wrappers.

C++
#include <list>
#include <forward_list>

// Doubly Linked List
std::list<int> dll = {10, 20};
dll.push_back(30);
dll.push_front(5);

// Singly Linked List
std::forward_list<int> sll = {1, 2};
sll.push_front(0); // only front inserts
JAVA
import java.util.LinkedList;

// Standard Doubly-Linked List
LinkedList<Integer> list = new LinkedList<>();

list.addLast(20);
list.addFirst(10);

// Mutation tracking
list.removeFirst();

PYTHON
# Standard list is dynamic array.
# For high performance links, use deque
from collections import deque

linked_nodes = deque([10, 20])
linked_nodes.append(30)     # right push
linked_nodes.appendleft(5)  # left push

linked_nodes.popleft()      # O(1) removal
