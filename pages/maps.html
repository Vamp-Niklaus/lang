Maps / Dictionaries / Key-Value

Associative tracking containers mapping unique keys to data blocks.

C++
#include <unordered_map>
#include <map>

// Hash table based (Unordered)
std::unordered_map<std::string, int> hashM;
hashM["id"] = 101;

// Red-Black Tree Based (Sorted Key)
std::map<int, std::string> sortedM;

// Check key existence
if(hashM.find("id") != hashM.end()) { ... }
JAVA
import java.util.HashMap;
import java.util.TreeMap;

// Hash Table variant
HashMap<String, Integer> map = new HashMap<>();
map.put("id", 101);

// Sorted Variant
TreeMap<Integer, String> tree = new TreeMap<>();

// Check status
if(map.containsKey("id")) {
    int val = map.get("id");
}
JAVASCRIPT
// Map class instance
const hash = new Map();
hash.set("id", 101);

// Check presence & read
if(hash.has("id")) {
    let val = hash.get("id");
}

// Plain Object Alternative
const fallbackObj = { "id": 101 };
let alternativeVal = fallbackObj["id"];
PYTHON
# Native Dictionary
dict_map = {"id": 101}
dict_map["status"] = "Active"

# Verification check syntax
if "id" in dict_map:
    val = dict_map["id"]

# Safer inline retrieval fallback
val = dict_map.get("missing_key", 0)
