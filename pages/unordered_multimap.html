Unordered Multimap

Hash table map that allows duplicate keys.

Category: Unordered Container

Overview

#include <unordered_map>
std::unordered_multimap<int, std::string> umm;

umm.insert({1, "One"});
umm.insert({1, "Uno"});

// operator[] is not available
auto range = umm.equal_range(1);

HashMap<K, List<V>>

Java requires manually mapping a Key to an ArrayList of Values.

import java.util.HashMap;
import java.util.ArrayList;
import java.util.List;

HashMap<String, List<Integer>> multiMap = new HashMap<>();

// O(1) Insert
multiMap.putIfAbsent("A", new ArrayList<>());
multiMap.get("A").add(1);
multiMap.get("A").add(2);

// O(1) Get all values
List<Integer> vals = multiMap.getOrDefault("A", new ArrayList<>());
for (int v : vals) {
    System.out.println(v);
}

collections.defaultdict(list)

Automatically initializes a list for new keys.

from collections import defaultdict

d = defaultdict(list)
d['A'].append(1)
d['A'].append(2)

Map of Arrays

Use a JavaScript Map where the value is an array.

const multiMap = new Map();

function insert(key, val) {
    if (!multiMap.has(key)) {
        multiMap.set(key, []);
    }
    multiMap.get(key).push(val);
}

function getValues(key) {
    return multiMap.get(key) || [];
}

insert("A", 1);
insert("A", 2);

console.log(getValues("A")); // [1, 2]
