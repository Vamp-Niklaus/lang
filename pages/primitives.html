Primitive Types & Declarations

Comparing variables, types, and baseline mutation syntax.

C++ Static / Compiled 
// Basic primitives
int age = 22;
float gpa = 8.7f;
double pi = 3.14159;
char letter = 'A';
bool isDev = true;

// String (Standard Library)
#include <string>
std::string name = "Rakesh";

// Constant value
const int MAX_USERS = 100;

// Type inference (C++11+)
auto count = 5; // Deduced as int
JAVA Static / JVM 
// Basic primitives
int age = 22;
float gpa = 8.7f;
double pi = 3.14159;
char letter = 'A';
boolean isDev = true;

// String (Object class)
String name = "Rakesh";


// Constant value
final int MAX_USERS = 100;

// Local variable inference
var count = 5; // Java 10+
JAVASCRIPT Dynamic / JIT 
// Dynamic typing primitives
let age = 22;        // Number
let gpa = 8.7;       // Number
let name = "Rakesh"; // String
let isDev = true;    // Boolean

// Big Integer
let huge = 9007199254740991n;

// Constants
const MAX_USERS = 100;

// Missing items markers
let unassigned;      // undefined
let clearValue = null;
PYTHON Dynamic / Interp 
# Purely dynamically bound
age = 22             # int
gpa = 8.7            # float
name = "Rakesh"      # str
is_dev = True        # bool (Capitalized)

# Constant (By convention only)
MAX_USERS = 100

# Type Hinting syntax (Optional)
count: int = 5
title: str = "Developer"

# Empty value type
empty_val = None
