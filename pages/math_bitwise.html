Math & Bitwise

Mathematical functions, bit manipulation, and numeric gotchas.

// MATH ()
int a = std::abs(-5);
double p = std::pow(2, 3); // 8.0
double s = std::sqrt(16);  // 4.0
int g = std::gcd(10, 5);   // ()

// MIN/MAX ()
int m = std::max(a, b);

// BITWISE
int x = 5;       // 0101
int and_ = x & 1; // 0001
int or_ = x | 2;  // 0111
int xor_ = x ^ 1; // 0100
int ls = x << 1;  // 1010 (x * 2)
int rs = x >> 1;  // 0010 (x / 2)

// BUILTIN POPCOUNT (Count set bits)
int bits = __builtin_popcount(x); // GCC specific!
// MATH (java.lang.Math)
int a = Math.abs(-5);
double p = Math.pow(2, 3); // 8.0
double s = Math.sqrt(16);  // 4.0

// MIN/MAX
int m = Math.max(a, b);

// GCD (Write your own!)

// POPCOUNT (Write your own)
function popcount(n) {
    let count = 0;
    while(n) { n &= (n - 1); count++; }
    return count;
}
